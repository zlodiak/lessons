echo $SHELL 
bash 
echo $EDITOR 
не установлено в среде 
export EDITOR=/path/editor (для bash) 
echo $EDITOR 
/path/editor 
mc -> F9 -> ... -> использовать внешний редактор который берется из 
переменной среды $EDITOR 

[unix1]~ > echo $SHELL 
/usr/local/bin/bash 
[unix1]~ > echo $EDITOR 
/usr/bin/vi 
[unix1]~ > grep use_internal_edit .mc/ini 
use_internal_edit=0 
[unix1]~ >
