(function() { var addEvent, title = document.querySelectorAll('.title'), i = 0, max = title.length; if (document.addEventListener) { addEvent = function(el, type, handler) { el.addEventListener(type, handler, false) }; } else { addEvent = function(el, type, handler) { el.attachEvent("on" + type, handler) }; }; for (; i < max; i += 1) { addEvent(title[i], "mouseover", function(e) { e = e || window.event; var inner = e.target.nextElementSibling; inner.style.display = "block"; }); addEvent(title[i], "mouseout", function(e) { e = e || window.event; var inner = e.target.nextElementSibling; inner.style.display = "none"; }); }; })();
